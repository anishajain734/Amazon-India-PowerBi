**Dax Formula Used in the PowerBi Report**
1. % Contribution by Payment Method = DIVIDE([Total Sales],CALCULATE([Total Sales],ALL ( 'Amazon 2025'[Payment_Method])))
2. AOV = DIVIDE([Total Sales],DISTINCTCOUNT('Amazon 2025'[Order_ID]))
3. Avg Review Rating = AVERAGE('Amazon 2025'[Review_Rating])
4. Best Category = TOPN(1,VALUES('Amazon 2025'[Product Category]),[Total Sales],DESC)
5. Category Rank = RANKX(ALL('Amazon 2025'[Product Category]),[Total Sales],,DESC)
6. Customer Orders = CALCULATE(DISTINCTCOUNT('Amazon 2025'[Order_ID]),ALLEXCEPT('Amazon 2025', 'Amazon 2025'[Customer_ID]))
7. Customer Type = IF([Customer Orders] > 1, "Repeat", "New")
8. OnTime % = DIVIDE(CALCULATE(COUNT('Amazon 2025'[Order_ID]),'Amazon 2025'[Delivery_Status] = "Delivered"),COUNT('Amazon 2025'[Order_ID]))
9. Rolling 3M Sales = CALCULATE([Total Sales],DATESINPERIOD('Amazon 2025'[Date],MAX('Amazon 2025'[Date]),-3,MONTH))
10. Sales MoM % = VAR PrevMonth = CALCULATE([Total Sales],DATEADD('Amazon 2025'[Date], -1, MONTH))RETURNDIVIDE([Total Sales] - PrevMonth, PrevMonth)
11. Sales MoM Display = VAR MoM = [Sales MoM %]RETURNIF (MoM > 0,"▲ +" & FORMAT ( MoM, "0%" ),IF (MoM < 0, "▼ " & FORMAT ( MoM, "0%" ),"0%"))
12. Selected State = "State Details: " & SELECTEDVALUE('Amazon 2025'[State],"Multiple States")
13. Total Quantity Sold = SUM('Amazon 2025'[Quantity])
14. Total Sales = SUM('Amazon 2025'[Total Sales])
